DECLARE
N NUMBER;
PROCEDURE FACTORIAL(N NUMBER)IS
F NUMBER:=1;
BEGIN
FOR I IN 1..N LOOP
F:=F*I;
END LOOP;
DBMS_OUTPUT.PUT_LINE(F);
END;
BEGIN
FACTORIAL(5);
END;
/


DECLARE
A NUMBER;
B NUMBER;
C NUMBER;
FUNCTION FINDMAX(X IN NUMBER,Y IN NUMBER)
RETURN NUMBER IS
Z NUMBER;
BEGIN
IF X>Y THEN
Z:=X;
ELSE
Z:=Y;
END IF;
RETURN Z;
END;
BEGIN
A:=25;
B:=15;
C:=FINDMAX(A,B);
DBMS_OUTPUT.PUT_LINE('MAX OF TWO NO IS:-'||C);
END;
/
